./src/main.ts template:

import express, { Request, Response } from "express";
import cors from "cors";

const PORT = 8080;

const app = express();
app.use(express.json())
app.use(cors());

app.listen(PORT, () => {
  console.log(`Server is running on port ${PORT}`);
});
